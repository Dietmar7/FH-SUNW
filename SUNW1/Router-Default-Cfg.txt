! Router Default-Konfiguration für SUNW1
! As-is-template for reference purposes, does not include details as specific IP addressing

enable		! Wechseln Sie in den Privilege-Exec Modus
conf t		! Wechseln Sie in den Konfigurations-Modus

hostname IMA-RO-<Platz-Nr>
ip domain name ima.local

crypto key generate rsa mod 2048    	! Key erzeugen für SSH

username admin priv 15 secret 0 admin123  ! User für ssh login
enable secret 0 cisco

!
line vty 0 15
login local					! Lokale Userauthentifizierung

logging buffered 256000 debugging	

interface gig0/0/0
ip address 172.30.1.4<Platz-Nr> 255.255.0.0  
no shutdown

interface gig0/0/1
ip address 192.168.<UE-Nr><Platz-Nr>.1 255.255.255.0 
no shutdown

ip dhcp pool LAN			! DHCP für eigenen Arbeitsplatz-PC
network 192.168.<UE-Nr><Platz-Nr>.0 255.255.255.0 
default-router 192.168.<UE-Nr><Platz-Nr>.1		
dns-server 172.30.255.253 
end

write mem! Dauerhaftes speichern der Konfiguration im NVRAM


! Löschen der Konfig nach Übung
write erase
![confirm] -> Enter/yes
